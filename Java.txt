#include <stdio.h>
int main(){
    int n,p[20],c;
    printf("Enter the number of pages: ");
    scanf("%d",&n);
    printf("Enter the page numbers: ");
    for(int i=0;i<n;i++){
        scanf("%d",&p[i]);
    }
    printf("Enter the number of frames: ");
    scanf("%d",&c);
    int m[c],t[c],fl=0,cl=0;
    for(int i=0;i<c;i++){
        m[i]=-1,t[i]=0;
    }
    printf("\nStep-by-step execution:\n");
    for(int i=0;i<n;i++){
        int pa=p[i],f=0;
        for(int j=0;j<c;j++){
            if(m[j]==pa){
                f=1;
                t[j]=++cl;
                break;
            }
        }
        if(!f){
           fl++;
           int lru=0;
           for(int j=1;j<c;j++){
            if(t[j]<t[lru]){
                lru=j;
            }
           }
           m[lru]=pa;
           t[lru]=++cl;
        }
        printf("Page %2d -> ",pa);
        for(int j=0;j<c;j++){
            if(m[j]==-1) printf(" - ");
            else printf("%2d ",m[j]);
        }
        printf("%s\n", f?"Hit":"Fault");
    }
    printf("\nTotal Page Faults: %d\n",fl);
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
int main() {
    int n,req[50],head,total=0,i;
    float avg;
    printf("Enter number of requests: ");
    scanf("%d",&n);
    printf("Enter request queue: ");
    for(i=0;i<n;i++)
        scanf("%d",&req[i]);
    printf("Enter initial head position: ");
    scanf("%d",&head);
    printf("\nMovement: %d",head);
    for (i =0;i< n;i++){
        total+=abs(req[i]-head);
        head=req[i];
        printf(" -> %d",head);
    }
    avg=(float)total/n;
    printf("\nTotal Head Movement = %d\n",total);
    printf("\nAverage Head Movement = %.2f\n",avg);
    return 0;
}

#include <stdio.h>
int main(){
    int t;
    printf("Enter total number of blocks: ");
    scanf("%d",&t);
    int a[200]={0};
    int s,l,i,al;
    printf("How many blocks are already allocated? ");
    scanf("%d",&al);
    printf("Enter already allocated block numbers: ");
    for(i=0;i<al;i++){
        int b;
        scanf("%d",&b);
        if (b<0||b>=t){
            printf("Invalid block number %d\n",b);
            return 0;
        }
        a[b]=1;
    }
    printf("Enter starting block of file: ");
    scanf("%d",&s);
    printf("Enter length of file: ");
    scanf("%d",&l);
    if(s<0||s+l>t){
        printf("\nAllocation out of range!\n");
        return 0;
    }
    int f=0;
    for(i=s;i<s+l;i++){
        if(a[i] == 0)
            f++;
    }
    if(f==l){
        printf("\nFile allocated successfully!\nBlocks: ");
        for(i=s;i<s+l;i++){
            printf("%d ",i);
            a[i]=1;
        }
        printf("\n");
    }else{
        printf("\nFile cannot be allocated (block already used)\n");
    }
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
int main(){
    int n,t[20],s[20],c[20],h,hu[20],i,ch,r=1;
    printf("DINING PHILOSOPHER PROBLEM\n");
    printf("Enter total philosophers (max 20): ");
    scanf("%d",&n);
    if(n<=0||n>20){
        printf("Invalid number!\n");
        return 0;
    }
    for(i=0;i<n;i++){
        t[i]=i+1;s[i]=1;c[i]=0;
    }
    printf("How many hungry: ");
    scanf("%d",&h);
    if(h<0||h>n){
        printf("Invalid number!\n");
        return 0;
    }
    if(h==n){
        printf("All hungry. Deadlock may occur!\n");
        return 0;
    }
    for(i=0;i<h;i++){
        printf("Enter hungry philosopher %d: ",i+1);
        scanf("%d",&hu[i]);
        if(hu[i]<1||hu[i]>n){
                printf("Invalid!\n");
                return 0;
        }
        s[hu[i]-1]=2;
    }
    printf("How many can eat at same time (1 or 2)? ");
    scanf("%d",&ch);
    if(ch<1||ch>2){
        printf("Invalid choice!\n");
        return 0;
    }
    printf("\n--- Simulation Starting ---\n");
    int rem=h;
    while(rem>0){
        int ec=0;
        for(i=0;i<h&&ec<ch;i++){
            int p=hu[i]-1,l=p,rgt=(p+1)%n;
            if(s[p]==2&&c[l]==0&&c[rgt]==0){
                c[l]=1;c[rgt]=1;
                printf("Philosopher %d is hungry... waiting\n",t[p]);
                sleep(1);
                printf("Philosopher %d starts eating\n",t[p]);
                s[p]=3;ec++;
            }
        }
        sleep(2);
        for(i=0;i<h;i++){
            int p=hu[i]-1;
            if(s[p]==3){
                printf("Philosopher %d finished eating\n",t[p]);
                s[p]=1;
                c[p]=0;
                c[(p+1)%n]=0;
                rem--;
            }
        }
        printf("\n--- Current Status ---\n");
        for(i=0;i<n;i++){
            if(s[i]==1) printf("Philosopher %d thinking\n",t[i]);
            else if(s[i]==2) printf("Philosopher %d hungry\n",t[i]);
            else printf("Philosopher %d eating\n",t[i]);
        }
        printf("\n");
    }
    printf("--- Simulation Completed ---\n");
    return 0;
}
